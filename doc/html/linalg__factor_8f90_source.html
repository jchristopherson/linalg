210  subroutine form_lu_all(lu, ipvt, u, p, err)
211  ! Arguments
212  real(dp), intent(inout), dimension(:,:) :: lu
213  integer(i32), intent(in), dimension(:) :: ipvt
214  real(dp), intent(out), dimension(:,:) :: u, p
215  class(errors), intent(inout), optional, target :: err
216 
217  ! Local Variables
218  integer(i32) :: j, jp, n, flag
219  class(errors), pointer :: errmgr
220  type(errors), target :: deferr
221  character(len = 128) :: errmsg
222 
223  ! Parameters
224  real(dp), parameter :: zero = 0.0d0
225  real(dp), parameter :: one = 1.0d0
226 
227  ! Initialization
228  n = size(lu, 1)
229  if (present(err)) then
230  errmgr => err
231  else
232  errmgr => deferr
233  end if
234 
235  ! Input Check
236  flag = 0
237  if (size(lu, 2) /= n) then
238  flag = 1
239  else if (size(ipvt) /= n) then
240  flag = 2
241  else if (size(u, 1) /= n .or. size(u, 2) /= n) then
242  flag = 3
243  else if (size(p, 1) /= n .or. size(p, 2) /= n) then
244  flag = 4
245  end if
246  if (flag /= 0) then
247  ! One of the input arrays is not sized correctly
248  write(errmsg, '(AI0A)') "Input number ", flag, &
249  " is not sized correctly."
250  call errmgr%report_error("form_lu_all", trim(errmsg), &
251  la_array_size_error)
252  return
253  end if
254 
255  ! Ensure P starts off as an identity matrix
256  call dlaset('A', n, n, zero, one, p, n)
257 
258  ! Process
259  do j = 1, n
260  ! Define the pivot matrix
261  jp = ipvt(j)
262  if (j /= jp) call swap(p(j,1:n), p(jp,1:n))
263 
264  ! Build L and U
265  u(1:j,j) = lu(1:j,j)
266  u(j+1:n,j) = zero
267 
268  if (j > 1) lu(1:j-1,j) = zero
269  lu(j,j) = one
270  end do
271  end subroutine

693  subroutine form_qr_no_pivot(r, tau, q, work, olwork, err)
694  ! Arguments
695  real(dp), intent(inout), dimension(:,:) :: r
696  real(dp), intent(in), dimension(:) :: tau
697  real(dp), intent(out), dimension(:,:) :: q
698  real(dp), intent(out), target, dimension(:), optional :: work
699  integer(i32), intent(out), optional :: olwork
700  class(errors), intent(inout), optional, target :: err
701 
702  ! Parameters
703  real(dp), parameter :: zero = 0.0d0
704 
705  ! Local Variables
706  integer(i32) :: j, m, n, mn, qcol, istat, flag, lwork
707  real(dp), pointer, dimension(:) :: wptr

713

716  n = size(r, 2)
717  mn = min(m, n)

723  end if

728  flag = 2
729  else if (size(q, 1) /= m .or. (qcol /= m .and. qcol /= n)) then
730  flag = 3

732  flag = 3
733  end if

744  call dorgqr(m, qcol, mn, q, m, tau, temp, -1, flag)
745  lwork = int(temp(1), i32)
746  if (present(olwork)) then
747  olwork = lwork
748  return
749  end if

751  ! Local Memory Allocation
752  if (present(work)) then
753  if (size(work) < lwork) then
754  ! ERROR: WORK not sized correctly
755  call errmgr%report_error("form_qr_no_pivot", &
