566  function mat_qr(a, pvt) result(x)
567  ! Arguments
568  real(real64), intent(in), dimension(:,:) :: a
569  logical, intent(in), optional :: pvt
570  type(qr_results) :: x
571 
572  ! Local Variables
573  logical :: use_pivot
574  integer(int32) :: m, n, mn
575  integer(int32), allocatable, dimension(:) :: jpvt
576  real(real64), allocatable, dimension(:) :: tau
577 
578  ! Memory Allocation
579  use_pivot = .false.
580  if (present(pvt)) use_pivot = pvt
581  m = size(a, 1)
582  n = size(a, 2)
583  mn = min(m, n)
584  allocate(tau(mn))
585  allocate(x%q(m,m))
586  allocate(x%r(m,n))
587 
588  ! Compute the factorization, and then form Q, R, and P
589  x%r = a
590  if (use_pivot) then
591  allocate(x%p(n,n))
592  allocate(jpvt(n))
593  jpvt = 0 ! Ensure all columns are free columns
594  call qr_factor(x%r, tau, jpvt)
595  call form_qr(x%r, tau, jpvt, x%q, x%p)
596  else
597  call qr_factor(x%r, tau)
598  call form_qr(x%r, tau, x%q)
599  end if
600  end function
